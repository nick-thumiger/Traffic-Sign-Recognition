Traffic Sign Classifier¶

Convolutional Neural Network Approach¶

In the project below, a convolutional neural network is constructed to classify a series of German Traffic Signs.

Dataset Summary & Exploration¶

The pickled data is a dictionary with 4 key/value pairs:

  • 'features' is a 4D array containing raw pixel data of the traffic sign images, (num examples, width, height, channels).
  • 'labels' is a 1D array containing the label/class id of the traffic sign. The file signnames.csv contains id -> name mappings for each id.
  • 'sizes' is a list containing tuples, (width, height) representing the original width and height the image.
  • 'coords' is a list containing tuples, (x1, y1, x2, y2) representing coordinates of a bounding box around the sign in the image.

Model Architecture Design¶

Pre-process the Data Set (normalization, grayscale, etc.)¶

Minimally, the image data should be normalized so that the data has mean zero and equal variance. For image data, (pixel - 128)/ 128 is a quick way to approximately normalize the data.

The data is also converted into grayscale.

In [5]:
import cv2

def preprocess_img(imgset):
    # Make the images in the training set all grayscale
    outset = []
    for ind in range(len(imgset)):
        outset.append(cv2.cvtColor(imgset[ind], cv2.COLOR_RGB2GRAY))
    outset = np.array(outset)
    outset = np.asarray(outset, dtype=int)[...,None]
    
    # Randomise and Normalise the image pixel data
    outset = np.float32((outset + -128)/128)
    
    return outset

X_processed = preprocess_img(X_train)
X_valid = preprocess_img(X_valid)

Model Architecture¶

A basic LeNet architecture was used to solve this problem, with further optimisations and improvements listed below.

  1. A dropout layer is included after both fully connected layers to avoid overfitting
  2. The images are converted to grayscale
  3. The images are normalized to a mean of 0 (ranging from 0-1)

import tensorflow as tf
from tensorflow.contrib.layers import flatten

def LeNet(x):    
    # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer
    mu = 0
    sigma = 0.1
    
    # Layer 1: Convolutional. Input = 32x32x3. Output = 28x28x6.
    conv_layer1W = tf.Variable(tf.truncated_normal([5,5,1,6], mean=mu, stddev=sigma))
    conv_layer1B = tf.Variable(tf.zeros(6))
    conv_layer1 = tf.nn.conv2d(x, conv_layer1W, strides=[1,1,1,1], padding='VALID') + conv_layer1B

    # Activation.
    conv_layer1Act = tf.nn.relu(conv_layer1)

    # Pooling. Input = 28x28x6. Output = 14x14x6.
    conv_layer1Act = tf.nn.max_pool(conv_layer1Act, ksize=[1,2,2,1], strides=[1,2,2,1], padding='VALID')
    
    # Layer 2: Convolutional. Output = 10x10x16.
    conv2_W = tf.Variable(tf.truncated_normal([5,5,6,16], mean=mu, stddev=sigma))
    conv2_B = tf.Variable(tf.zeros(16))
    conv2 = tf.nn.conv2d(conv_layer1Act, conv2_W, strides=[1,1,1,1], padding='VALID') + conv2_B
    
    # Activation.
    conv2 = tf.nn.relu(conv2)

    # Pooling. Input = 10x10x16. Output = 5x5x16.
    conv2 = tf.nn.max_pool(conv2, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')

    # Flatten. Input = 5x5x16. Output = 400.
    fc0   = flatten(conv2)
    
    # Layer 3: Fully Connected. Input = 400. Output = 120.
    fc1_W = tf.Variable(tf.truncated_normal(shape=(400, 120), mean = mu, stddev = sigma))
    fc1_b = tf.Variable(tf.zeros(120))
    fc1   = tf.matmul(fc0, fc1_W) + fc1_b
    
    # Activation.
    fc1    = tf.nn.relu(fc1)
    
    # Dropout layer for fully connected layer 1
    if TEST == 0:
        fc1 = tf.nn.dropout(fc1, KEEP_PROB)

    # Layer 4: Fully Connected. Input = 120. Output = 84.
    fc2_W  = tf.Variable(tf.truncated_normal(shape=(120, 84), mean = mu, stddev = sigma))
    fc2_b  = tf.Variable(tf.zeros(84))
    fc2    = tf.matmul(fc1, fc2_W) + fc2_b
    
    # Activation.
    fc2    = tf.nn.relu(fc2)
    
    # Dropout layer for fully connected layer 2
    if TEST == 0:
        fc2 = tf.nn.dropout(fc2, KEEP_PROB)

    # Layer 5: Fully Connected. Input = 84. Output = n_classes.
    fc3_W  = tf.Variable(tf.truncated_normal(shape=(84, n_classes), mean = mu, stddev = sigma))
    fc3_b  = tf.Variable(tf.zeros(n_classes))
    logits = tf.matmul(fc2, fc3_W) + fc3_b
    
    return logits

correct_prediction = tf.equal(tf.argmax(logits, 1), tf.argmax(one_hot_y, 1))
accuracy_operation = tf.reduce_mean(tf.cast(correct_prediction, tf.float32))
saver = tf.train.Saver()

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

    sess.run(tf.global_variables_initializer())

    for i in range(EPOCHS):
        X_processed, y_train = shuffle(X_processed, y_train)

            end = offset + BATCH_SIZE
            batch_x, batch_y = X_processed[offset:end], y_train[offset:end]

        validation_accuracy = evaluate(X_valid, y_valid)
        print("EPOCH {} ...".format(i+1))
        print("Validation Accuracy = {:.3f}".format(validation_accuracy))
        test_accuracy = evaluate(X_processed, y_train)
        print("Train Accuracy = {:.3f}".format(test_accuracy))
        print()
        
    saver.save(sess, './TrainingData/lenet')
    print("Model saved")
